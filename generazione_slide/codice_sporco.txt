/* Inizializzo il lock per gestire l'accesso alla risorsa condivisa. */ private final StampedLock lock = new StampedLock(); private double balance; public void readBalance() { long stamp = lock.tryOptimisticRead(); /* Provo una lettura senza bloccare nessuno. */ double currentBalance = balance; if (!lock.validate(stamp)) { /* Se qualcuno ha scritto nel frattempo, scatto al lock pessimistico. */ stamp = lock.readLock(); try { currentBalance = balance; } finally { lock.unlockRead(stamp); } } System.out.println("Balance letto: " + currentBalance); }
